import cv2
import pyautogui
import numpy as np
import time


def sol_tiklama():
    pyautogui.mouseDown(button='left')
    time.sleep(.8)  # Sol tıklama işlemi 1 saniye boyunca sürecek
    pyautogui.mouseUp(button='left')


def sol_tiklama2():
    pyautogui.mouseDown(button='left')


def sol_tiklama3():
    pyautogui.mouseUp(button='left')
    time.sleep(1.2)


def metni_ara(dosya_yolu, mesaj):
    yazi_goruntu = cv2.imread(dosya_yolu, cv2.IMREAD_GRAYSCALE)

    ekran_goruntu = np.array(pyautogui.screenshot())
    ekran_goruntu_gri = cv2.cvtColor(ekran_goruntu, cv2.COLOR_BGR2GRAY)

    sonuc = cv2.matchTemplate(ekran_goruntu_gri, yazi_goruntu, cv2.TM_CCOEFF_NORMED)
    _, max_val, _, max_loc = cv2.minMaxLoc(sonuc)

    eşik_değeri = 0.71

    if max_val > eşik_değeri:
        print(mesaj)
        return True
    else:
        return False


def main():
    interval_minutes = 20  # 20 dakika aralıklarla çalışacak
    next_trigger = time.time() + interval_minutes * 60

    while True:
        current_time = time.time()

        if current_time >= next_trigger:
            next_trigger = current_time + interval_minutes * 60

            pyautogui.press('a')  # 'a' tuşuna bas
            time.sleep(0.1)
            pyautogui.press('d')
            print("Tuşa basıldı!")

        # 1. metni ara ve tıklama yap
        dosya_yolu_1 = 'C:/Users/nos/Desktop/ekrangg/1.png'
        mesaj_1 = "1. görsel algılandı!"
        if metni_ara(dosya_yolu_1, mesaj_1):
            sol_tiklama()
            time.sleep(0.1)

        # 2. görsel algılandığında işlemi yap
        dosya_yolu_2 = 'C:/Users/nos/Desktop/ekrangg/2.png'
        mesaj_2 = "2. görsel algılandı!"
        if metni_ara(dosya_yolu_2, mesaj_2):
            sol_tiklama2()


        # 3. görsel algılandığında işlemi yap
        dosya_yolu_33 = 'C:/Users/nos/Desktop/ekrangg/a.png'
        mesaj_33 = "3. görsel algılandı!"
        if metni_ara(dosya_yolu_33, mesaj_33):
            sol_tiklama3()
            time.sleep(0.5)
            while True:
                if metni_ara(dosya_yolu_33, mesaj_33):
                    sol_tiklama3()
                if metni_ara(dosya_yolu_1, mesaj_1):
                    break
                else:
                    sol_tiklama2()

        time.sleep(0.5)


if __name__ == '__main__':
    main()
